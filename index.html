Supabase Setup
Supabase এ নিচের SQL run করো Table তৈরি করতে (Auth + Super Admin সহ):
create table if not exists attendance (
  id bigint primary key,
  user_id uuid references auth.users(id) default auth.uid(),
  owner_email text,
  name text, roll text, phone text, org text,
  type text, schedule_status text, range_status text,
  lat float8, lng float8, address text, ip text,
  notes text, photo text, agent text,
  timestamp timestamptz default now(),
  time_display text
);

alter table attendance add column if not exists owner_email text;

alter table attendance enable row level security;

drop policy if exists "select own" on attendance;
drop policy if exists "insert own" on attendance;
drop policy if exists "update own" on attendance;
drop policy if exists "delete own" on attendance;
drop policy if exists "select own or admin" on attendance;
drop policy if exists "update own or admin" on attendance;
drop policy if exists "delete own or admin" on attendance;

create policy "select own or admin" on attendance for select
  using (auth.uid() = user_id or auth.jwt() ->> 'email' = 'samirhossain5a@gmail.com');
create policy "insert own" on attendance for insert
  with check (auth.uid() = user_id);
create policy "update own or admin" on attendance for update
  using (auth.uid() = user_id or auth.jwt() ->> 'email' = 'samirhossain5a@gmail.com');
create policy "delete own or admin" on attendance for delete
  using (auth.uid() = user_id or auth.jwt() ->> 'email' = 'samirhossain5a@gmail.com');

create table if not exists broadcast (
  id bigint primary key default 1,
  message text,
  updated_at timestamptz default now(),
  updated_by text
);
insert into broadcast (id, message) values (1, '') on conflict (id) do nothing;
alter table broadcast enable row level security;
drop policy if exists "broadcast select all" on broadcast;
drop policy if exists "broadcast update admin only" on broadcast;
create policy "broadcast select all" on broadcast for select using (true);
create policy "broadcast update admin only" on broadcast for update
  using (auth.jwt() ->> 'email' = 'samirhossain5a@gmail.com');
